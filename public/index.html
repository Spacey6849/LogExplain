import requests

# LogExplain API Client
API_URL = "https://log-explain.vercel.app/v1/logs/explain"
API_KEY = "YOUR_API_KEY"

response = requests.post(
    API_URL,
    headers={
        "Content-Type": "application/json",
        "x-api-key": API_KEY,
    },
    json={
        "log": "2025-01-15 ERROR [PostgreSQL] Connection refused - max connections reached"
    }
)

result = response.json()
print(f"Severity: {result['severity']['score']}/100")
print(f"Summary: {result['explanation']['summary']}")
print(f"Fix: {result['explanation']['recommendedActions']}")

package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "net/http"
)

func main() {
    apiURL := "https://log-explain.vercel.app/v1/logs/explain"
    apiKey := "YOUR_API_KEY"

    body, _ := json.Marshal(map[string]string{
        "log": "2025-01-15 ERROR [PostgreSQL] Connection refused - max connections reached",
    })

    req, _ := http.NewRequest("POST", apiURL, bytes.NewBuffer(body))
    req.Header.Set("Content-Type", "application/json")
    req.Header.Set("x-api-key", apiKey)

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil { fmt.Println(err); return }
    defer resp.Body.Close()

    var result map[string]interface{}
    json.NewDecoder(resp.Body).Decode(&result)
    fmt.Printf("Result: %+v\n", result)
}
